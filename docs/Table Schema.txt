CREATE TABLE alumni (
    alumni_id INT AUTO_INCREMENT NOT NULL,
    first_name VARCHAR(255) NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    branch VARCHAR(255) NOT NULL,
    graduation_year VARCHAR(255) NULL,
    company_name VARCHAR(255) NOT NULL,
    designation VARCHAR(255) NULL,
    ctc VARCHAR(255) NULL,
    phone_number VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    social_link VARCHAR(255) NULL,
    PRIMARY KEY (alumni_id)
);


CREATE TABLE company (
    company_id INT AUTO_INCREMENT NOT NULL,
    name VARCHAR(255) NOT NULL,
    role VARCHAR(255) NOT NULL,
    ctc VARCHAR(255) NOT NULL,
    job_description TEXT NULL,
    logo_url VARCHAR(255) NULL,
    web_url VARCHAR(255) NULL,
    allow_branch VARCHAR(255) NULL,
    require_skills VARCHAR(255) NULL,
    location VARCHAR(255) NULL,
    PRIMARY KEY (company_id)
);


CREATE TABLE users (
    user_id INT AUTO_INCREMENT NOT NULL,
    first_name VARCHAR(255) NOT NULL,
    last_name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    user_type VARCHAR(255) NOT NULL,
    PRIMARY KEY (user_id)
);
